带回调的confirm
code
//带回调函数的confirm调用方式
$.confirm('这是confirm',function(a){
	$.alert('你点击了'+(a?'确定':'取消'))
})
return false阻止对话框关闭
code
//回调函数中的this指向对话框（jquery对象）
//return false可以阻止对话框关闭
$.confirm('点击确定不会关闭对话框！！', function(e) {
	if(!e) return;
	this.find('.alert_content').html('回调函数阻止了对话框关闭，点击取消才能关闭！！');
	return false; 
})
